#include <iostream>

using namespace std;

int main() {
	int i,j,a[1000000],k,max,n;
	cin>>n>>k;
	
	for(i=0;i<n;i++)
	{   cin>>a[i];  }
	
	
    for (int i = 0; i <= n - k ; i++)  
    {  
        max = a[i];  
  
        for (j = 1; j < k; j++)  
        {  
            if (a[i + j] > max)  
                max = a[i + j];  
        }  
        cout << max << " ";  
    }  
  
	
	
	
	return 0;
}
