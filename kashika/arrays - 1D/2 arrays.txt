#include <iostream>

using namespace std;

int main() {
	int n,i,sum1=0,sum2=0,cnt1=0,cnt2=0;
    cin>>n;
    int arr1[n],arr2[n];
    for(i=0;i<n;i++)
        cin>>arr1[i];
   
   for(i=0;i<n;i++)
        cin>>arr2[i];
   
   i=0;
   
   while(i<n)
    {
        if(arr1[i]!=-1)
            sum1+=arr1[i];
        else
            cnt1+=1;
        if(arr2[i]!=-1)
            sum2+=arr2[i];
        else
            cnt2+=1;
      i++;
    }
   
   if(cnt1==1 && cnt2==1)
    {
        cout<<"Infinite";
    }
   
   else if(cnt1==2 || cnt2==2)
    {
      if(cnt1==2)           
        {
            if(sum1>sum2)
               cout<<"0";
         else
           cout<<sum2-sum1+1;
        }  
       else
        {
            
        if(sum2>sum1)
           cout<<"0";
         else  
         cout<<sum1-sum2+1;
    }}
    else
    {
         if(cnt1==1)
        {
            if(sum2<sum1)
                cout<<"0"<<endl;
            else
                cout<<"1"<<endl;
        }
        else
        {
            if(sum1<sum2)
                cout<<"0"<<endl;
            else
                cout<<"1"<<endl;
        }
    }
 
 
}
